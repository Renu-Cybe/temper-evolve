## Phase 3: 增强心跳集成 (P1)

### 目标
实现事件驱动的四自系统，将心跳与审计、持久化系统链接

### 核心实现

#### 1. 事件发布集成

**`temper/heartbeat/evolver.py`** 新增事件发布：

```python
# 自感知
async def _self_check(self):
    # 发布健康检查事件
    event_bus.publish(Event(
        type=EventType.HEALTH_CHECK,
        source="TemperEvolver",
        data={'check_id': self._stats['self_checks']}
    ))
    
    # 告警时发布
    event_bus.publish(Event(
        type=EventType.ALERT_TRIGGERED,
        source="TemperEvolver",
        data={'alerts': alerts, 'summary': summary}
    ))

# 自适应
async def _adapt(self):
    # 发布参数调优事件
    event_bus.publish(Event(
        type=EventType.PARAMETER_TUNED,
        source="TemperEvolver",
        data={'suggestions_count': len(suggestions)}
    ))

# 自编译
async def _repair_check(self):
    # 发布代码扫描事件
    event_bus.publish(Event(
        type=EventType.CODE_GENERATED,
        source="TemperEvolver",
        data={'scan_issues': len(issues)}
    ))

# 系统生命周期
def start(self):
    event_bus.publish(Event(
        type=EventType.SYSTEM_START,
        source="TemperEvolver",
        data={'version': '3.0.0', 'config': self.config.__dict__}
    ))

def stop(self):
    event_bus.publish(Event(
        type=EventType.SYSTEM_STOP,
        source="TemperEvolver",
        data={'stats': self._stats, 'duration_seconds': duration}
    ))
```

#### 2. 持久化集成

```python
def get_stats(self):
    stats = {...}
    
    # 持久化统计（新增）
    if hasattr(self.system, 'state') and self.system.state:
        self.system.state.set("evolver.stats", stats)
        self.system.state.set("evolver.last_updated", datetime.now().isoformat())
```

#### 3. 审计集成

所有四自操作记录审计日志：
- 健康检查 → `AuditCategory.HEALTH`
- 参数调优 → `AuditCategory.ADAPTATION`
- 代码扫描 → `AuditCategory.REPAIR`
- 系统启停 → `AuditCategory.SYSTEM`

#### 4. 事件处理器注册

```python
def register_evolver_event_handlers(event_bus, system):
    """注册进化器事件处理器
    
    将四自系统事件链接到审计和持久化系统
    """
    def on_health_check(event): ...
    def on_alert_triggered(event): ...
    def on_parameter_tuned(event): ...
    def on_code_generated(event): ...
    def on_system_start(event): ...
    def on_system_stop(event): ...
    
    # 注册所有处理器
    event_bus.subscribe(EventType.HEALTH_CHECK, on_health_check)
    event_bus.subscribe(EventType.ALERT_TRIGGERED, on_alert_triggered)
    # ...
```

#### 5. 主系统集成

**`temper.py`**:
```python
from temper.core.events import event_bus
from temper.heartbeat import register_evolver_event_handlers

def start_evolver(self):
    self.evolver = TemperEvolver(self.four_self_system, config)
    
    # 注册事件处理器（新增）
    register_evolver_event_handlers(event_bus, self.four_self_system)
    
    self.evolver.start()
```

```python
@dataclass
class HeartbeatConfig:
    """心跳进化器配置（Phase 4 新增）"""
    enabled: bool = True
    debug: bool = False
    
    # 自感知
    self_check_interval: int = 60
    health_report_interval: int = 300
    
    # 自适应
    adapt_interval: int = 300
    adapt_threshold: float = 0.1
    
    # 自组织
    workflow_check_interval: int = 60
    
    # 自编译

    auto_repair_enabled: bool = False
```

#### 3. 配置转换方法

**`temper/heartbeat/evolver.py`**:
```python
@staticmethod
def from_heartbeat_config(config: HeartbeatConfig) -> 'EvolverConfig':
    """从主配置系统创建 EvolverConfig"""
    return EvolverConfig(
        self_check_interval=config.self_check_interval,
        health_report_interval=config.health_report_interval,
        adapt_interval=config.adapt_interval,
        # ...
    )

def to_heartbeat_config(self) -> dict:
    """转换为字典格式"""
    return {
        'enabled': self.enabled,
        'debug': self.debug,
        'self_check_interval': self.self_check_interval,
        # ...
    }
```
